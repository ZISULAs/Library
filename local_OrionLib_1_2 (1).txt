local OrionLib = loadstring(game:HttpGet("https://raw.githubusercontent.com/soluesQP/Orion/main/Modded/Best"))()
local Player = game.Players.LocalPlayer
local Window = OrionLib:MakeWindow({
    Name = "QP_태권도",
    HidePremium = false,
    SaveConfig = true,
    IntroText = "QP_태권도 "
})

OrionLib:MakeNotification({
    Name = "로그인! (V1)",
    Content = "당신은 다음 계정으로 로그인했습니다. : "..Player.Name.."  ",
    Image = "rbxassetid://4483345998",
    Time = 3
})

local Tab = Window:MakeTab({
    Name = "메인",
    Icon = "rbxassetid://4483345998",
    PremiumOnly = false
})

function w()
  OrionLib:MakeNotification({
	Name = "작동됨!  ✅️",
	Content = ""..Player.Name.."님 감사합니다!!",
	Image = "rbxassetid://4483345998",
	Time = 5
})
end

local expToggle = false
local expToggle4x = false
local teleportToggle = false

Tab:AddToggle({
    Name = "EXP 오토 팜",
    Default = false,
    Callback = function(value)
        expToggle = value
        if expToggle then
            while expToggle do
                local args = {
                    [1] = 1
                }
                game:GetService("ReplicatedStorage"):WaitForChild("Cilcker"):FireServer(unpack(args))
                wait(0.0001)
              w()
            end
        end
    end
})

Tab:AddToggle({
    Name = "돈 도우미 (클릭은 별도)",
    Default = false,
    Callback = function(value)
        teleportToggle = value
        if teleportToggle then
            while teleportToggle do
                local function teleportPlayer(pos)
                    local player = game.Players.LocalPlayer
                    if player then
                        player.Character.HumanoidRootPart.CFrame = CFrame.new(pos)
                    end
                end

                teleportPlayer(Vector3.new(70.05, 2, 225.65))
                wait(2)
                teleportPlayer(Vector3.new(0.59, 4, 47.97))
                wait(2)
            end
        end
    end
})
